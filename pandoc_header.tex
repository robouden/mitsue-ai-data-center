\usepackage{fancyhdr}
\usepackage{lastpage}
\usepackage{xcolor}
\definecolor{headercolor}{RGB}{80,80,80}

\pagestyle{fancy}
\fancyhf{}

% Header: title left, last modified date right
\fancyhead[L]{\small\color{headercolor}\textit{$if(title)$$title$$endif$}}
\fancyhead[R]{\small\color{headercolor}Last modified: $date$}

% Footer: page X of Y centered
\fancyfoot[C]{\small\color{headercolor}Page \thepage\ of \pageref*{LastPage}}

\renewcommand{\headrulewidth}{0.4pt}
\renewcommand{\footrulewidth}{0.4pt}

% Ensure first page also has header/footer
\fancypagestyle{plain}{
  \fancyhf{}
  \fancyhead[L]{\small\color{headercolor}\textit{$if(title)$$title$$endif$}}
  \fancyhead[R]{\small\color{headercolor}Last modified: $date$}
  \fancyfoot[C]{\small\color{headercolor}Page \thepage\ of \pageref*{LastPage}}
  \renewcommand{\headrulewidth}{0.4pt}
  \renewcommand{\footrulewidth}{0.4pt}
}
